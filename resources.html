quick_start.py 
# Install: pip install sesmine
import sesmine as sm

# Authenticate
client = sm.Client(api_key="YOUR_API_KEY")

# Get live commodity prices
prices = client.prices.get_spot(
    commodities=["gold", "copper", "lithium"]
)
print(prices.gold.spot)   # 3142.50

# Get AI price forecast
forecast = client.ai.price_forecast(
    commodity="gold",
    horizon=90,
    scenario="base"
)
print(forecast.target)    # 3380.00
print(forecast.confidence) # 0.82

# Stream live prices via WebSocket
async with client.stream.prices() as ws:
    async for tick in ws:
        print(tick.commodity, tick.price)
